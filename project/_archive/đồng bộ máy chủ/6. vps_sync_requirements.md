## 📦 API Endpoints Cần Tạo

### 1. Product Units (Đơn vị tính)
```
POST /api/sync/product-units
```
**Payload**:
```json
{
  "license_key": "xxx",
  "data": [
    {
      "id": 1,
      "product_id": 5,
      "unit_name": "Thùng",

      "price": 120000.0,
      "barcode": "8938502001234"

}
```

### 7. Product Transactions (Thẻ Kho) ⭐ NEW
```
POST /api/sync/product-transactions
```
**Payload**:
```json
{
  "license_key": "xxx",
  "data": [
    {
      "id": 1,
      "product_id": 5,
      "transaction_type": "IN",
      "quantity": 100.0,
      "reference_type": "PURCHASE",
      "reference_id": 15,
      "note": "Nhập hàng - PO #123",
      "balance_after": 150.0,
      "created_at": "2026-01-31T10:00:00Z"
    },
    {
      "id": 2,
      "product_id": 5,
      "transaction_type": "OUT",
      "quantity": 20.0,
      "reference_type": "SALE",
      "reference_id": 45,
      "note": "Bán hàng - INV-1706770800",
      "balance_after": 130.0,
      "created_at": "2026-01-31T11:00:00Z"
    }
  ]
}
```

**Transaction Types**:
- `IN` = Nhập kho (from purchase)
- `OUT` = Xuất kho (from sale)
- `ADJUST` = Điều chỉnh (manual)

## 🗄️ Database Schema Cần Tạo

```sql
-- 1. Product Units
CREATE TABLE sync_product_units (
    id BIGINT, store_id UUID,
    product_id BIGINT, unit_name TEXT,
    exchange_value DECIMAL, price DECIMAL, barcode TEXT,
    PRIMARY KEY (store_id, id)
);

-- 2. Store Funds
CREATE TABLE sync_store_funds (
    store_id UUID PRIMARY KEY,
    current_balance DECIMAL,
    updated_at TIMESTAMPTZ
);

-- 3. Cash Transactions
CREATE TABLE sync_cash_transactions (
    id BIGINT, store_id UUID,
    transaction_type TEXT, amount DECIMAL,
    description TEXT, reference_type TEXT, reference_id BIGINT,
    created_at TIMESTAMPTZ,
    PRIMARY KEY (store_id, id)
);

-- 4. Product Batches
CREATE TABLE sync_product_batches (
    id BIGINT, store_id UUID,
    product_id BIGINT, purchase_item_id BIGINT,
    expiry_date DATE, import_date DATE,
    quantity DECIMAL, remaining_quantity DECIMAL,
    PRIMARY KEY (store_id, id)
);

-- 5. Payment Vouchers
CREATE TABLE sync_payment_vouchers (
    id BIGINT, store_id UUID,
    voucher_number TEXT, recipient TEXT,
    amount DECIMAL, payment_method TEXT, note TEXT,
    created_at TIMESTAMPTZ,
    PRIMARY KEY (store_id, id)
);

-- 6. Store Settings
CREATE TABLE sync_store_settings (
    store_id UUID PRIMARY KEY,
    name TEXT, address TEXT, phone TEXT,
    security_pin TEXT, updated_at TIMESTAMPTZ
);

-- 7. Product Transactions (NEW)
CREATE TABLE sync_product_transactions (
    id BIGINT, store_id UUID,
    product_id BIGINT, transaction_type TEXT,
    quantity DECIMAL, reference_type TEXT, reference_id BIGINT,
    note TEXT, balance_after DECIMAL,
    created_at TIMESTAMPTZ,
    PRIMARY KEY (store_id, id)
);
```
